EasyExcel 百万级数据导入导出 Java 解决方案

1. 核心依赖配置

EasyExcel 的核心优势在于其低内存占用和流式处理能力。需要在 Maven 项目中引入以下依赖：

Maven (pom.xml):

<dependency>
    <groupId>com.alibaba</groupId>
    <artifactId>easyexcel</artifactId>
    <!-- 建议使用最新稳定版本，这里以 3.3.3 为例 -->
    <version>3.3.3</version>
</dependency>

2. 百万级数据处理最佳实践

处理百万级数据，核心思想是流式处理和分批操作，避免一次性将所有数据加载到内存中，从而防止内存溢出（OOM）。

功能 核心策略 EasyExcel 实现方式 关键点
数据导入 (Read) SAX 事件驱动解析 + 分批入库 使用 EasyExcel.read() 方法，并传入自定义的 ReadListener。 在 invoke() 方法中接收数据，积累到一定数量（如 1000 条）后，批量写入数据库。
数据导出 (Write) 分页查询 + 分批写入 使用 ExcelWriter 对象，配合 write() 方法进行多次写入。 每次从数据库分页查询一批数据（如 10 万条），然后调用 write() 写入，直到所有数据写完。

3.2. 百万级数据导入 (Read)

导入的关键是实现 ReadListener，并在其中进行分批处理和入库操作。

import com.alibaba.excel.context.AnalysisContext;
import com.alibaba.excel.read.listener.ReadListener;
import java.util.ArrayList;
import java.util.List;

public class ExcelDataListener implements ReadListener<ExcelData> {

    /**
     * 每隔 1000 条数据存储一次，然后清理 list，方便内存回收
     */
    private static final int BATCH_COUNT = 1000;
    private List<ExcelData> cachedDataList = new ArrayList<>(BATCH_COUNT);

    // 假设这是数据库服务
    // private UserService userService;

    // 构造函数可以传入 Service
    // public ExcelDataListener(UserService userService) {
    //     this.userService = userService;
    // }

    /**
     * 每一条数据解析都会调用
     */
    @Override
    public void invoke(ExcelData data, AnalysisContext context) {
        cachedDataList.add(data);
        // 达到 BATCH_COUNT 就进行数据存储操作
        if (cachedDataList.size() >= BATCH_COUNT) {
            saveData();
            // 存储完成清理 list
            cachedDataList = new ArrayList<>(BATCH_COUNT);
        }
    }

    /**
     * 所有数据解析完成都会调用
     */
    @Override
    public void doAfterAllAnalysed(AnalysisContext context) {
        // 确保最后剩余的数据也被存储
        saveData();
        System.out.println("所有数据解析完成！");
    }

    /**
     * 批量存储数据到数据库
     */
    private void saveData() {
        System.out.println(cachedDataList.size() + " 条数据，开始批量存储...");
        // 实际应用中，这里调用 Service 层进行批量插入
        // userService.batchInsert(cachedDataList);
        System.out.println("存储成功！");
    }
}

3.3. 百万级数据导出 (Write)

导出的关键是使用 ExcelWriter 进行流式写入，并结合数据库的分页查询。

import com.alibaba.excel.EasyExcel;
import com.alibaba.excel.ExcelWriter;
import com.alibaba.excel.write.metadata.WriteSheet;
import java.io.OutputStream;
import java.util.List;

public class ExportService {

    // 假设这是数据库服务
    // private UserService userService;

    /**
     * 百万级数据导出方法
     * @param outputStream 写入流
     */
    public void exportExcel(OutputStream outputStream) {
        // 1. 创建 ExcelWriter 对象
        try (ExcelWriter excelWriter = EasyExcel.write(outputStream, ExcelData.class).build()) {

            // 2. 创建一个 Sheet
            WriteSheet writeSheet = EasyExcel.writerSheet("用户数据").build();

            // 3. 分页查询并分批写入
            int pageSize = 100000; // 每次查询 10 万条
            long totalCount = 1000000; // 假设总共有 100 万条数据
            long totalPage = (totalCount + pageSize - 1) / pageSize;

            for (int pageNum = 0; pageNum < totalPage; pageNum++) {
                // 模拟数据库分页查询
                // List<ExcelData> dataList = userService.queryDataByPage(pageNum, pageSize);
                List<ExcelData> dataList = generateMockData(pageSize, pageNum);

                // 4. 写入数据
                excelWriter.write(dataList, writeSheet);
                System.out.println("已写入第 " + (pageNum + 1) + " 页数据，共 " + dataList.size() + " 条。");
            }

        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    // 模拟生成数据的方法
    private List<ExcelData> generateMockData(int size, int page) {
        List<ExcelData> list = new ArrayList<>(size);
        long startId = (long) page * size + 1;
        for (int i = 0; i < size; i++) {
            ExcelData data = new ExcelData();
            data.setId(startId + i);
            data.setUsername("user_" + (startId + i));
            data.setEmail("user_" + (startId + i) + "@example.com");
            list.add(data);
        }
        return list;
    }
}
